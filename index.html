3. Subscription Lock Controls & UNEB Engine

Automate multi-tenant billing protections and calculate grading components, aggregates, and division criteria under standard UNEB rules.

-- Subscription verification constraint trigger logic
create or replace function public.check_school_subscription_standing()
returns trigger as $$
declare
    current_school_status subscription_status_type;
    is_school_suspended boolean;
begin
    select subscription_status, is_suspended 
    into current_school_status, is_school_suspended
    from public.schools where id = new.school_id;

    if (current_school_status != 'ACTIVE' or is_school_suspended = true) then
        raise exception 'Operation Denied: The subscription for this school has expired or has been suspended. Access to data entry, report rendering, and processing modules is restricted until outstanding arrears are cleared.';
    end if;
    return new;
end;
$$ language plpgsql;

create trigger enforce_subscription_on_marks_entry
    before insert or update or delete on public.marks_entries
    for each row execute function public.check_school_subscription_standing();

create trigger enforce_subscription_on_student_add
    before insert or update or delete on public.students
    for each row execute function public.check_school_subscription_standing();

create or replace function public.calculate_uneb_subject_grade(numeric_mark numeric)
returns varchar as $$
begin
    if numeric_mark >= 80.00 then return 'D1';
    elsif numeric_mark >= 75.00 then return 'D2';
    elsif numeric_mark >= 70.00 then return 'C3';
    elsif numeric_mark >= 65.00 then return 'C4';
    elsif numeric_mark >= 60.00 then return 'C5';
    elsif numeric_mark >= 55.00 then return 'C6';
    elsif numeric_mark >= 50.00 then return 'P7';
    elsif numeric_mark >= 45.00 then return 'P8';
    else return 'F9';
    end if;
end;
$$ language plpgsql immutable;

create or replace function public.process_and_compile_termly_uneb_results(
    target_student_id uuid,
    target_term_id uuid
)
returns void as $$
declare
    v_school_id uuid;
    v_total_marks numeric(6,2);
    v_avg_mark numeric(5,2);
    v_aggregate_sum integer := 0;
    v_division uneb_division_type := 'FAIL';
    v_credit_count integer := 0;
    v_p8_count integer := 0;
    v_math_grade varchar(2);
    v_eng_grade varchar(2);
begin
    select school_id into v_school_id from public.students where id = target_student_id;

    select sum(total_calculated_mark), avg(total_calculated_mark)
    into v_total_marks, v_avg_mark
    from public.marks_entries where student_id = target_student_id and term_id = target_term_id;

    select count(*) filter (where grade_achieved in ('D1','D2','C3','C4','C5','C6')) into v_credit_count
    from public.marks_entries where student_id = target_student_id and term_id = target_term_id;

    select count(*) filter (where grade_achieved in ('P7','P8')) into v_p8_count
    from public.marks_entries where student_id = target_student_id and term_id = target_term_id;

    select grade_achieved into v_math_grade from public.marks_entries 
    where student_id = target_student_id and term_id = target_term_id 
      and subject_id in (select id from public.subjects where code ilike 'UCE-MAT%' or name ilike '%Mathematics%') limit 1;

    select grade_achieved into v_eng_grade from public.marks_entries 
    where student_id = target_student_id and term_id = target_term_id 
      and subject_id in (select id from public.subjects where code ilike 'UCE-ENG%' or name ilike '%English%') limit 1;

    select sum(sub.agg_val) into v_aggregate_sum
    from (
        select case 
            when grade_achieved = 'D1' then 1 when grade_achieved = 'D2' then 2
            when grade_achieved = 'C3' then 3 when grade_achieved = 'C4' then 4
            when grade_achieved = 'C5' then 5 when grade_achieved = 'C6' then 6
            when grade_achieved = 'P7' then 7 when grade_achieved = 'P8' then 8
            else 9 end as agg_val
        from public.marks_entries where student_id = target_student_id and term_id = target_term_id
        order by agg_val asc limit 8
    ) sub;

    -- Strict traditional routing classification algorithm rules
    if (v_aggregate_sum <= 32 and v_credit_count >= 6 and v_math_grade not in ('F9','P8','P7') and v_eng_grade not in ('F9','P8','P7')) then
        v_division := 'DIVISION_1';
    elsif (v_aggregate_sum <= 45 and v_credit_count >= 4) then
        v_division := 'DIVISION_2';
    elsif (v_aggregate_sum <= 58 and v_credit_count >= 3) then
        v_division := 'DIVISION_3';
    elsif (v_aggregate_sum <= 72 and (v_credit_count >= 1 or v_p8_count >= 3)) then
        v_division := 'DIVISION_4';
    else
        v_division := 'FAIL';
    end if;

    insert into public.termly_summaries (
        school_id, student_id, term_id, total_marks, average_mark, total_aggregates, division_achieved, updated_at
    ) values (
        v_school_id, target_student_id, target_term_id, v_total_marks, v_avg_mark, v_aggregate_sum, v_division, now()
    ) on conflict (student_id, term_id) do update set
        total_marks = excluded.total_marks,
        average_mark = excluded.average_mark,
        total_aggregates = excluded.total_aggregates,
        division_achieved = excluded.division_achieved,
        updated_at = now();
end;
$$ language plpgsql;

import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from "https://esm.sh/@supabase/supabase-js@2.38.4"

const ULTRAMSG_INSTANCE_ID = Deno.env.get('ULTRAMSG_INSTANCE_ID')!;
const ULTRAMSG_API_TOKEN = Deno.env.get('ULTRAMSG_API_TOKEN')!;
const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;
const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!;

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });

  try {
    const { phoneNumber } = await req.json();
    const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);
    const generatedOtp = Math.floor(100000 + Math.random() * 900000).toString();
    const expiryWindow = new Date(Date.now() + 5 * 60 * 1000).toISOString();

    const { error: dbError } = await supabase
      .from('whatsapp_otp_codes')
      .insert([{ phone_number: phoneNumber, otp_code: generatedOtp, expires_at: expiryWindow }]);

    if (dbError) throw dbError;

    const messagePayload = `Your Multi-School Portal security validation key token is: *${generatedOtp}*. Access window closes in 5 minutes.`;
    
    await fetch(`https://api.ultramsg.com/${ULTRAMSG_INSTANCE_ID}/messages/chat`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({ token: ULTRAMSG_API_TOKEN, to: phoneNumber, body: messagePayload, priority: '10' })
    });

    return new Response(JSON.stringify({ success: true }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, status: 200 });
  } catch (err: any) {
    return new Response(JSON.stringify({ error: err.message }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, status: 400 });
  }
});

"use client";
import React, { useEffect, useState } from 'react';
import { createClientComponentClient } from '@supabase/auth-helpers-nextjs';
import { useRouter } from 'next/navigation';

export default function TenantDashboardLayout({ children }: { children: React.ReactNode }) {
  const supabase = createClientComponentClient();
  const router = useRouter();
  const [tenant, setTenant] = useState<any>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function evaluateTenantContext() {
      const { data: { session } } = await supabase.auth.getSession();
      if (!session) return router.push('/login');

      const schoolId = session.user.user_metadata.school_id;
      const { data } = await supabase.from('schools').select('*').eq('id', schoolId).single();
      setTenant(data);
      setLoading(false);
    }
    evaluateTenantContext();
  }, [supabase, router]);

  return (

          <span className="truncate">{tenant?.name}</span>
        </div>
        <nav className="flex-1 p-4 space-y-1 text-sm text-slate-300">
          <a href="/dashboard/students" className="block p-3 rounded hover:bg-slate-800">Student Master Indexes</a>
          <a href="/dashboard/academics" className="block p-3 rounded hover:bg-slate-800">Grades Compilation Ledger</a>
          <a href="/dashboard/billing" className="block p-3 rounded hover:bg-slate-800">Bursary & Accounts Balance</a>
        </nav>
      </aside>
      <div className="flex-1 flex flex-col">
        {tenant?.subscription_status !== 'ACTIVE' && (
          <div className="bg-red-600 text-white text-center p-2.5 text-xs font-bold uppercase tracking-wider">
            System Lock Active: Functions are locked until current standing dues are cleared.
          </div>
        )}
        <main className="p-8 flex-1">{children}</main>
      </div>
    </div>
  );
}

6. Flutter Mobile Client Architecture

Tenancy Client Orchestrator Header (tenant_api_client.dart)

import 'package:dio/dio.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';

class TenantApiClient {
  final Dio _dio = Dio();
  final _secureStorage = const FlutterSecureStorage();

  TenantApiClient() {
    _dio.options.baseUrl = "https://your-supabase-id.supabase.co/rest/v1";
    _dio.options.connectTimeout = const Duration(seconds: 15);

    _dio.interceptors.add(InterceptorsWrapper(
      onRequest: (options, handler) async {
        final token = await _secureStorage.read(key: "auth_jwt_token");
        final schoolId = await _secureStorage.read(key: "tenant_school_id");
        
        if (token != null) {
          options.headers["Authorization"] = "Bearer $token";
          options.headers["apikey"] = "YOUR_SUPABASE_ANON_KEY";
          options.headers["X-Tenant-School-ID"] = schoolId;
        }
        return handler.next(options);
      },
      onError: (DioException error, handler) {
        if (error.response?.statusCode == 403) {
          print("Subscription Error Exception Block: Data execution path context locked.");
        }
        return handler.next(error);
      }
    ));
  }
  Dio get instance => _dio;
}
